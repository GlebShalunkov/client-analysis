import pandas as pd
import numpy as np
from datetime import datetime, timedelta

# Создаём данные о клиентах
np.random.seed(42)

clients = ['Анна', 'Евгений', 'Александр', 'Мария', 'Дмитрий', 
           'Елена', 'Сергей', 'Ольга', 'Иван', 'Татьяна',
           'Павел', 'Наталья', 'Алексей', 'Юлия', 'Артём']

projects = ['Анализ продаж', 'Дашборд для CEO', 'SQL оптимизация', 
            'Парсинг сайта', 'Отчёт для инвесторов', 'ETL пайплайн',
            'Визуализация данных', 'Прогноз оттока', 'A/B тест анализ',
            'Дата-аудит']

data = []
for i in range(50):  # 50 проектов
    client = np.random.choice(clients)
    project = np.random.choice(projects)
    amount = np.random.choice([15000, 25000, 35000, 50000, 75000, 100000])
    status = np.random.choice(['завершён', 'в работе', 'оплачен'], p=[0.6, 0.3, 0.1])
    
    # Дата проекта (последние 6 месяцев)
    days_ago = np.random.randint(0, 180)
    date = (datetime.now() - timedelta(days=days_ago)).strftime('%Y-%m-%d')
    
    data.append({
        'client': client,
        'project': project,
        'amount': amount,
        'status': status,
        'date': date,
        'rating': np.random.randint(1, 6)  # Оценка клиента (1-5)
    })

df = pd.DataFrame(data)
df.to_csv('client_projects.csv', index=False)
print("📊 Данные о клиентах созданы:")

# Считаем метрики по каждому клиенту
client_stats = df.groupby('client').agg({
    'amount': ['sum', 'mean', 'count'],
    'rating': 'mean'
}).round(0)

client_stats.columns = ['total_spent', 'avg_project', 'projects_count', 'avg_rating']
client_stats = client_stats.reset_index()

# 1. Ранжирование по общей сумме (кто больше заплатил)
client_stats['rank_total'] = client_stats['total_spent'].rank(
    method='dense', ascending=False
).astype(int)

# 2. Ранжирование по среднему чеку
client_stats['rank_avg'] = client_stats['avg_project'].rank(
    method='dense', ascending=False
).astype(int)

# 3. Ранжирование по количеству проектов (лояльность)
client_stats['rank_loyalty'] = client_stats['projects_count'].rank(
    method='dense', ascending=False
).astype(int)

# 4. Ранжирование по рейтингу (как клиент оценивает)
client_stats['rank_rating'] = client_stats['avg_rating'].rank(
    method='dense', ascending=False
).astype(int)

# Топ-10 по общей сумме
print("\n💰 ТОП-10 КЛИЕНТОВ ПО ОБЩЕЙ СУММЕ:")
top_by_money = client_stats.nlargest(10, 'total_spent')
for i, row in top_by_money.iterrows():
    print(f"{row['rank_total']:2}. {row['client']:10} | "
          f"Сумма: {row['total_spent']:6.0f}₽ | "
          f"Проектов: {row['projects_count']:2.0f} | "
          f"Рейтинг: {row['avg_rating']:.1f}⭐")

# Задача: Найти клиентов, которые заказывали 2+ проекта подряд
# (Аналог Consecutive Numbers)

print("\n" + "="*60)
print("🔄 КЛИЕНТЫ С ПОВТОРНЫМИ ЗАКАЗАМИ ПОДРЯД")
print("="*60)

# Сортируем по дате
df['date'] = pd.to_datetime(df['date'])
df_sorted = df.sort_values(['client', 'date'])

# Добавляем предыдущий и следующий заказ
df_sorted['prev_date'] = df_sorted.groupby('client')['date'].shift(1)
df_sorted['next_date'] = df_sorted.groupby('client')['date'].shift(-1)
df_sorted['days_since_prev'] = (df_sorted['date'] - df_sorted['prev_date']).dt.days

# Считаем количество заказов подряд (в пределах 30 дней)
df_sorted['consecutive'] = (df_sorted['days_since_prev'] <= 30).astype(int)

# Находим клиентов с 2+ заказами подряд
repeat_clients = df_sorted[df_sorted['consecutive'] == 1].groupby('client').size()
repeat_clients = repeat_clients[repeat_clients >= 1].sort_values(ascending=False)

print("\n👥 Клиенты с повторными заказами (в течение 30 дней):")
for client, count in repeat_clients.items():
    print(f"  • {client}: {count+1} заказов подряд")
